#the short and long windows
short_window = 40
long_window = 100

# Initialize the `decision` DataFrame with the `decision` column
decisions = pd.DataFrame(index=analysis_data.index)
decisions['signal'] = 0.0

# Create short simple moving average 
decisions['short_mavg'] = analysis_data['close'].rolling(window=short_window, min_periods=1, center=False).mean()

# Create long simple moving average 
decisions['long_mavg'] = analysis_data['close'].rolling(window=long_window, min_periods=1, center=False).mean()

# Create decisions
decisions['signal'][short_window:] = np.where(decisions['short_mavg'][short_window:] 
                                            > decisions['long_mavg'][short_window:], 1.0, 0.0)   

# Generate trading orders
decisions['positions'] = decisions['signal'].diff()

# Initialize the plot figure
fig = plt.figure()

# Add a subplot and label for y-axis
ax1 = fig.add_subplot(111,  ylabel='Price in $')

# Plot the closing price
analysis_data['close'].plot(ax=ax1, color='r', lw=2.)

# Plot the short and long moving averages
decisions[['short_mavg', 'long_mavg']].plot(ax=ax1, lw=2.)

# Plot the buy signals
ax1.plot(decisions.loc[decisions.positions == 1.0].index, 
         decisions.short_mavg[decisions.positions == 1.0],
         '^', markersize=10, color='m')
         
# Plot the sell signals
ax1.plot(decisions.loc[decisions.positions == -1.0].index, 
         decisions.short_mavg[decisions.positions == -1.0],
         'v', markersize=10, color='k')
         
# Show the plot
plt.show()
